# For firmware build (AVR32)
cmake --preset mcu-build
cmake --build --preset mcu-build

# For host-side unit testing (Windows/MinGW)
cmake --preset windows-build
cmake --build --preset windows-build
ctest --preset windows-build

# For running Cppcheck
-> first navigate to build/windows_build directory
cmake --build . --target cppcheck

# For running gcov coverage tools
cmake --preset windows-coverage
cmake --build --preset windows-coverage
ctest --preset windows-build
-> navigate to build/windows_coverage directory
cmake --build . --target coverage
