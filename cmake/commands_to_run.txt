# For firmware build (AVR32)
cmake --preset mcu-build
cmake --build --preset mcu-build

# For host-side unit testing (Windows/MinGW)
cmake --preset windows-build
cmake --build --preset windows-build
ctest --preset windows-build

# For running CppCheck
-> first navigate to the Windows build directory build/windows_build, then run CMake "build" command to run CppCheck
cd 
cmake --build . --target cppcheck

# For running clang-format_sources
-> first navigate to the Windows build directory build/windows_build, then run CMake "build" command to run CppCheck
cmake --build . --target format_sources
-> formatted copies of source files will be in build/windows_build/clang-format-output
